# Javascript - If..Else

## If

~~~javascript
var notaAluno = 10;

if (notaAluno >= 6)
  console.log("Aluno aprovado");
~~~

## If..Else

~~~javascript
var notaAluno = 10;

if (notaAluno >= 6)
  console.log("Aluno aprovado");
else 
  console.log("Aluno reprovado");
~~~

## If..Else If..

~~~javascript
var notaAluno = 10;

if (notaAluno == 10)
  console.log("Aluno aprovado com louvor");
else if (notaAluno >= 6)
  console.log("Aluno aprovado");
else
  console.log("Aluno reprovado");
~~~
