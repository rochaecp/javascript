# Javascript - Comando if

## if

~~~javascript
var notaAluno = 10;

if (notaAluno >= 6)
    console.log("Aluno aprovado");
~~~

## if ... else

~~~javascript
var notaAluno = 10;

if (notaAluno >= 6)
    console.log("Aluno aprovado");
else 
    console.log("Aluno reprovado");
~~~

## if ... else if ... else

~~~javascript
var notaAluno = 10;

if (notaAluno == 10)
    console.log("Aluno aprovado com louvor");
else if (notaAluno >= 6)
    console.log("Aluno aprovado");
else
    console.log("Aluno reprovado");
~~~
