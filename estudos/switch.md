# Javascript - Switch

~~~javascript
const numeroMes = 2;

switch (numeroMes) {
    case 1:
        console.log("Janeiro");
        break;
    case 2:
        console.log("Fevereiro");
        break;
    default:
        console.log("Número inválido");
}
~~~

~~~javascript
const fruta = "pera";

switch (fruta) {
    case "banana":
        console.log("banana é bom");
        break;
    case "pera":
        console.log("pera é melhor");
        break;
    default:
        console.log("erro");
}
~~~
